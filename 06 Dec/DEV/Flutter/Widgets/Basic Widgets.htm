    return MaterialApp(
      title: 'SnackBar Example',
      home: Scaffold(
        appBar: AppBar(

        ),
        body: Center(
          child: ElevatedButton(
            child: Text('Show SnackBar'),
            onPressed: () {
              ScaffoldMessenger.of(context).showSnackBar(
                SnackBar(
                  content: Text('Hello, SnackBar!'),
                  duration: Duration(seconds: 3),
                ),
              );
            },
          ),
        ),
      ),
    );
  }
}

                        child: Text('OK'),

                          Navigator.of(context).pop();
                        },
                      ),
                    ],
                  );
                },
              );
            },
          ),
        ),
      ),
    );
  }
