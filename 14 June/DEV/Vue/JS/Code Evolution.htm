  </div>

<script>
import ChildComponent from './ChildComponent.vue';

export default {
  components: {
    ChildComponent
  },
  props: {
    message: String
  },
  methods: {
    updateMessage(newMessage) {
      this.message = newMessage;
    }
  }
}
</script>

<!-- ChildComponent.vue -->
<template>
  <div>
    {{ message }}
    <button @click="updateParentMessage">Update Message</button>
  </div>
</template>

<script>
export default {
  props: {
    message: String
  },
  methods: {
    updateParentMessage() {
      this.$emit('updateMessage', 'Message updated from ChildComponent.');
    }
  }
}
</script>
