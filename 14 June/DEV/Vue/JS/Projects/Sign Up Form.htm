<script>
import { ref } from "vue";

export default {
  name: "RegistrationForm",
  setup() {
    const name = ref("");
    const phone = ref("");
    const email = ref("");
    const password = ref("");
    const gender = ref("");
    const country = ref("");
    const agree = ref(false);

    const nameValid = ref(false);
    const phoneValid = ref(false);
    const emailValid = ref(false);
    const passwordValid = ref(false);
    const genderValid = ref(false);
    const countryValid = ref(false);
const agreeValid = ref(false);

// Validation functions
const validateName = () => {
  nameValid.value = name.value.trim() !== "";
};
const validatePhone = () => {
  const phoneRegex = /^[0-9]{10}$/;
  phoneValid.value = phoneRegex.test(phone.value);
};
const validateEmail = () => {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  emailValid.value = emailRegex.test(email.value);
};
const validatePassword = () => {
  passwordValid.value = password.value.length >= 8;
};
const validateGender = () => {
  genderValid.value = gender.value !== "";
};
const validateCountry = () => {
  countryValid.value = country.value !== "";
};
const validateAgree = () => {
  agreeValid.value = agree.value;
};

// Computed property for form validity
const formValid = computed(() => {
  return (
    nameValid.value &&
    phoneValid.value &&
    emailValid.value &&
    passwordValid.value &&
    genderValid.value &&
    countryValid.value &&
    agreeValid.value
  );
});

// Submit form function
const submitForm = () => {
  if (formValid.value) {
    // Do something with the form data, e.g. send to a server
    console.log("Form submitted!");
  } else {
    // Show error messages
    validateName();
    validatePhone();
    validateEmail();
    validatePassword();
    validateGender();
    validateCountry();
    validateAgree();
  }
};

// Watchers for validation
watch(name, validateName);
watch(phone, validatePhone);
watch(email, validateEmail);
watch(password, validatePassword);
watch(gender, validateGender);
watch(country, validateCountry);
watch(agree, validateAgree);

return {
  name,
  phone,
  email,
  password,
  gender,
  country,
  agree,
  nameValid,
  phoneValid,
  emailValid,
  passwordValid,
  genderValid,
  countryValid,
  agreeValid,
  formValid,
  submitForm,
};

},
};
</script>
