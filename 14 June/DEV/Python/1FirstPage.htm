Lambda Functions

def myFunc(a,b,c):
    return a*b+c/2
print(myFunc(3,4,5))      

myLambdaFunction = lambda a,b,c : a+b-c*2

print(myLambdaFunction(5,6,7))

Map Functions

#map(function, iretrables)

l1 = [2,3,4]
l2 = [5,6,7]
 
xyz = lambda n,m :n*m
 
l3 = map(xyz,l1,l2)
 
print(list(l3))

#filter(function,iterables)
numbers = [4,7,3,9,8,0,1]
def xyz(n):
    if n%2==0:
        return True
    else:
        return False
evenNumbers = filter(xyz,numbers)  
print(list(evenNumbers))
